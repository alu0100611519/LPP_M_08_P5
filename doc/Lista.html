def self.from_array(ary)
        ary.sort! { |x,y| x.dif <=> y.dif }
        head = Node.new(ary[0],nil,nil)
        previous = head
        current = Node.new(nil,nil,nil)
        ary[1..-1].each do |p|
                previous.sig = Node.new(p,nil,previous)
                current = previous.sig
                previous = current
        end
        Lista.new(head,current)
end
